⚙️ 통합 Google Sheets 설정

구글 스프레드시트 1개에 아래 Apps Script를 연결하면, 5개 게임의 랭킹이 시트별로 자동 분리 저장됩니다.

① Google Sheets 새 파일 생성 → 확장 프로그램 → Apps Script

아래 코드 전체를 붙여넣고 저장 후 배포 → 새 배포 → 웹 앱 → 액세스: 모든 사용자

const SHEETS = { acid: "1단계_산성비", memory: "2단계_기억의전당", debit: "3단계_차변대변", factory: "4단계_조립공장", flight: "5단계_플라이트" }; function getSheet(game) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const name = SHEETS[game] || game; return ss.getSheetByName(name) || ss.insertSheet(name); } function doPost(e) { const d = JSON.parse(e.postData.contents); const sh = getSheet(d.game || "unknown"); if (sh.getLastRow() === 0) sh.appendRow(["이름","점수","콤보","정답수","날짜","저장시각"]); sh.appendRow([ d.name, d.score, d.combo||0, d.correct||0, d.date||"", new Date().toLocaleString("ko-KR") ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const sh = getSheet(e.parameter.game || "unknown"); if (sh.getLastRow() <= 1) return ContentService.createTextOutput("[]") .setMimeType(ContentService.MimeType.JSON); const rows = sh.getDataRange().getValues().slice(1); const out = rows .map(r => ({name:r[0],score:+r[1],combo:+r[2],correct:+r[3],date:r[4]})) .sort((a,b) => b.score - a.score) .slice(0, 20); return ContentService .createTextOutput(JSON.stringify(out)) .setMimeType(ContentService.MimeType.JSON); }

② 배포 URL을 각 게임 파일 상단 SHEET_URL에 붙여넣기

5개 게임 파일 모두 동일한 URL 1개를 사용합니다. 각 게임은 game: "acid" 같은 식별자를 함께 전송해 시트를 자동 분리합니다.
